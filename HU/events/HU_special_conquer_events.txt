province_event = {
	id = HU_special_conquer_events.1
	title = "HU_special_conquer_events.1.title"
	desc = "HU_special_conquer_events.1.desc"
	picture = BATTLE_eventPicture
	is_triggered_only = yes
	trigger = {
		has_dlc = "Mandate of Heaven"
		NOT = {
			controlled_by = owner
		}
		OR = {
			AND = {
				owner = {
					OR = {
						has_reform = chinese_warlord
						has_reform = warlord_state
					}
				}
				from = {
					has_reform = celestial_empire
				}
			}
			AND = {
				owner = {
					OR = {
						has_reform = chinese_warlord
						has_reform = warlord_state
						has_reform = celestial_empire
					}
				}
				from = {
					OR = {
						has_reform = chinese_warlord
						has_reform = warlord_state
					}
				}
			}
		}
	}
	immediate = {
		hidden_effect = {
			set_province_flag = occupied_counting_flag
			remove_province_modifier = occupied_counting
			add_province_modifier = {
				name = "occupied_counting"
				duration = 120
			}
			controller = {
				country_event = {
					id = HU_special_conquer_events.2
				}
			}
		}
	}
	option = {
		name = "HU_special_conquer_events.1.a"
	}
}
